# Tabs

Component to render **tabs**.

The `Tabs` component can be configured to render the **tabs list** in various styles by changing `position` and `type` parameters.

## Usage

The Tabs component can be imported using

```jsx
import { Tabs } from '@tail-kit/tail-kit'
```

### Tabs with Icon

<Playground scope={{ HiUserCircle, HiOfficeBuilding, HiUsers, HiCreditCard }}>
  {function TabsWithIcon() {
    const tabsList = [
      {
        title: 'My Account',
        content: 'Account Details',
        icon: <HiUserCircle className="w-6 h-6" />,
        key: 'my-account',
      },
      {
        title: 'Company',
        content: 'Company Details',
        key: 'company',
        icon: <HiOfficeBuilding className="w-6 h-6" />,
      },
      {
        title: 'Team Members',
        content: 'Team Members Details',
        key: 'team-members',
        icon: <HiUsers className="w-6 h-6" />,
      },
      {
        title: 'Billing',
        content: 'Billing Details',
        key: 'billing',
        icon: <HiCreditCard className="w-6 h-6" />,
      },
    ]
    const tabs = tabsList.map((tab) => ({
      key: tab.key,
      title: tab.title,
      icon: tab.icon,
      content: (
        <>
          <h1 className="text-base font-semibold text-gray-800">{tab.title}</h1>
          <div className="text-sm text-gray-700">{tab.content}</div>
        </>
      ),
    }))
    return <Tabs tabs={tabs} />
  }}
</Playground>

<Playground scope={{ HiUserCircle, HiOfficeBuilding, HiUsers, HiCreditCard }}>
  {function TabsWithExtraContent() {
    const [activeTab, setActiveTab] = React.useState('my-account')
    const tabsList = [
      {
        title: 'My Account',
        content: 'Account Details',
        icon: <HiUserCircle className="w-6 h-6" />,
        key: 'my-account',
      },
      {
        title: 'Company',
        content: 'Company Details',
        key: 'company',
        icon: <HiOfficeBuilding className="w-6 h-6" />,
      },
      {
        title: 'Team Members',
        content: 'Team Members Details',
        key: 'team-members',
        icon: <HiUsers className="w-6 h-6" />,
      },
      {
        title: 'Billing',
        content: 'Billing Details',
        key: 'billing',
        icon: <HiCreditCard className="w-6 h-6" />,
      },
    ]
    const tabs = tabsList.map((tab) => ({
      key: tab.key,
      title: tab.title,
      icon: tab.icon,
      content: (
        <>
          <h1 className="text-base font-semibold text-gray-800">{tab.title}</h1>
          <div className="text-sm text-gray-700">{tab.content}</div>
        </>
      ),
    }))
    return (
      <Tabs
        tab={activeTab}
        onTabChange={setActiveTab}
        tabs={tabs}
        extraContent={
          <Select
            value={activeTab}
            onChange={setActiveTab}
            className="w-48"
            options={tabs.map((tab) => ({
              label: tab.title,
              value: tab.key,
              icon: tab.icon,
            }))}
            placeholder="Select Tab"
          />
        }
      />
    )
  }}
</Playground>

### Tabs Positions and Types

<Playground

  scope={{ HiUserCircle, HiOfficeBuilding, HiUsers, HiCreditCard }}

  {function TabsPositionAndTypes() {
    const [activeTab, setActiveTab] = React.useState('my-account')
    const [position, setPosition] = React.useState('top')
    const [tabStyle, setTabStyle] = React.useState('underline')
    const positionOptions = [
      { label: 'Top', value: 'top' },
      { label: 'Bottom', value: 'bottom' },
      { label: 'Left', value: 'left' },
      { label: 'Right', value: 'right' },
    ]
    const tabStyleOptions = [
      { label: 'Underline', value: 'underline' },
      { label: 'Pill', value: 'pill' },
    ]
    const tabsList = [
      {
        title: 'My Account',
        content: 'Account Details',
        icon: <HiUserCircle className="w-6 h-6" />,
        key: 'my-account',
      },
      {
        title: 'Company',
        content: 'Company Details',
        key: 'company',
        icon: <HiOfficeBuilding className="w-6 h-6" />,
      },
      {
        title: 'Team Members',
        content: 'Team Members Details',
        key: 'team-members',
        icon: <HiUsers className="w-6 h-6" />,
      },
      {
        title: 'Billing',
        content: 'Billing Details',
        key: 'billing',
        icon: <HiCreditCard className="w-6 h-6" />,
      },
    ]
    const tabs = tabsList.map((tab) => ({
      key: tab.key,
      title: tab.title,
      icon: tab.icon,
      content: (
        <>
          <h1 className="text-base font-semibold text-gray-800">{tab.title}</h1>
          <div className="text-sm text-gray-700">{tab.content}</div>
        </>
      ),
    }))
    return (
      <>
        <div className="flex items-center mb-6 text-sm">
          <p className="mr-4">Tabs Position:</p>
          <Radio.RadioGroup
            value={position}
            onChange={setPosition}
            options={positionOptions}
          />
          <div className="flex-1" />
          <p className="mr-4">Tab Style:</p>
          <Radio.RadioGroup
            value={tabStyle}
            onChange={setTabStyle}
            options={tabStyleOptions}
          />
        </div>
        <Tabs
          type={tabStyle}
          position={position}
          tab={activeTab}
          onTabChange={setActiveTab}
          tabs={tabs}
          extraContent={
            <Select
              value={activeTab}
              onChange={setActiveTab}
              className="w-48"
              options={tabs.map((tab) => ({
                label: tab.title,
                value: tab.key,
                icon: tab.icon,
              }))}
              placeholder="Select Tab"
            />
          }
        />
      </>
    )
  }}
