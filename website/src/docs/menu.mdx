  HiOutlineShare,

  HiOutlineChevronDown,

  HiOutlineDotsVertical,
  HiOutlineLogout,
  HiOutlineUserCircle,
  HiOutlineCog,
} from 'react-icons/hi'

# Menu

Component to render **dropdown menu**

Use `Menu.Item` and `Menu.Divider` components to render the drop down content

If `verticalPlacement` or `horizontalPlacement` is not provided, it would be computed based on the position of
trigger and menu content.

## Usage

The Menu component can be imported using

```jsx
import { Menu } from '@tail-kit/tail-kit'
```

### Dropdown Menu

<Playground
  scope={{
    HiOutlineShare,
    HiOutlinePencilAlt,
    HiOutlineTrash,
    HiOutlineChevronDown,
    HiOutlineChevronUp,
    HiOutlineChevronDown,
    HiOutlineChevronUp,
  }}
>
  {function DropdownMenu() {
    const menuContent = (
      <>
        <Menu.Item label="Share Blog" icon={<HiOutlineShare />} />
        <Menu.Item label="Edit Blog" icon={<HiOutlinePencilAlt />} />
        <Menu.Divider />
        <Menu.Item
          className="text-red-500"
          label="Delete Blog"
          icon={<HiOutlineTrash />}
        />
      </>
    )
    return (
      <div className="relative w-full h-80 all-menus-container">
        <div className="absolute top-0 left-0">
          <Menu trigger={<Button icon={<HiOutlineChevronDown />} />}>
            {menuContent}
          </Menu>
        </div>
        <div className="absolute bottom-0 left-0">
          <Menu trigger={<Button icon={<HiOutlineChevronUp />} />}>
            {menuContent}
          </Menu>
        </div>
        <div className="absolute top-0 right-0">
          <Menu trigger={<Button icon={<HiOutlineChevronDown />} />}>
            {menuContent}
          </Menu>
        </div>
        <div className="absolute bottom-0 right-0">
          <Menu trigger={<Button icon={<HiOutlineChevronUp />} />}>
            {menuContent}
          </Menu>
        </div>
      </div>
    )
  }}
</Playground>

### Menu within a Modal

<Playground
  scope={{
    HiOutlineChevronDown,
    HiOutlineShare,
    HiOutlinePencilAlt,
    HiOutlineTrash,
  }}
>
  {function MenuWithinAModal() {
    const [modalVisible, setModalVisible] = React.useState(false)
    return (
      <>
        <Button onClick={() => setModalVisible(true)}>
          Click To Open Modal
        </Button>
        <Modal
          title="Menu Within Modal"
          visible={modalVisible}
          onRequestClose={() => setModalVisible(false)}
          dividers
        >
          <div className="mb-4">
            Lorem ipsum dolor sit amet, consectetur adipiscing elit. Morbi
            gravida lorem nec velit egestas, quis varius mi condimentum. Nulla
            risus velit, imperdiet at pharetra a, eleifend eget massa. Donec
            porta purus eu ex hendrerit sagittis. Vestibulum eget mattis tellus,
            a dapibus nisi.
          </div>
          <Menu
            trigger={<Button icon={<HiOutlineChevronDown />}>Show Menu</Button>}
          >
            <Menu.Item label="Share Blog" icon={<HiOutlineShare />} />
            <Menu.Item label="Edit Blog" icon={<HiOutlinePencilAlt />} />
            <Menu.Divider />
            <Menu.Item
              className="text-red-500"
              label="Delete Blog"
              icon={<HiOutlineTrash />}
            />
          </Menu>
        </Modal>
      </>
    )
  }}
</Playground>
